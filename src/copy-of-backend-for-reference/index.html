async function sendToNanoBanana() { const faceFiles = document.getElementById('faceInput').files; const referenceFile = document.getElementById('referenceInput').files[0]; if (faceFiles.length === 0 || !referenceFile) { alert("Please select at least one face image and a reference image."); return; } const formData = new FormData(); for (let i = 0; i < faceFiles.length; i++) { formData.append('face_images', faceFiles[i]); } formData.append('reference_image', referenceFile); // Direct network call out to his Hugging Face Space endpoint const response = await fetch('https://stroudw-ref-angle2.hf.space/remix-face', { method: 'POST', body: formData }); const imageBlob = await response.blob(); const imageObjectURL = URL.createObjectURL(imageBlob); // Display the resulting modified image onto the web page document.getElementById('resultImage').src = imageObjectURL; }
